\section{The Prisoner's Dilemma}
% =============================================================================

\begin{frame}{The Prisoner's Dilemma}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{One-shot game.} Two players simultaneously choose to \alert{cooperate (C)} or \alert{betray (B)}.\\[0.8em]

      Payoffs satisfy \(T > R > P > S\):
      \begin{itemize}
        \item \(T = 4\) — temptation to defect
        \item \(R = 3\) — mutual cooperation
        \item \(P = 1\) — mutual defection
        \item \(S = 0\) — sucker's payoff
      \end{itemize}
      \vspace{0.6em}
      \textbf{Core tension.} Betraying is strictly dominant, yet \((B,B) = (1,1)\) is Pareto-dominated by \((C,C) = (3,3)\).
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \vspace{0.5em}
      \begin{tabular}{lcc}
        \toprule
                    & \C     & \B     \\
        \midrule
        \C          & (3,\,3) & (0,\,4) \\
        \B          & (4,\,0) & (1,\,1) \\
        \bottomrule
      \end{tabular}
      \vspace{0.4em}
      \captionof*{table}{Payoff matrix (row = Player 1)}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The Iterated Version (IPD)}
  \begin{itemize}
    \item Repetition changes the analysis: defection can now be \emph{punished}, cooperation \emph{rewarded}.
    \item Axelrod (1984) showed that \textbf{Tit for Tat} — cooperate first, then mirror — wins computer tournaments against a wide strategy set.
  \end{itemize}
  \vspace{0.8em}
  \begin{block}{Research question}
    Can a Q-learning agent learn an effective strategy in the IPD, and under what conditions does it converge to cooperation or defection?
  \end{block}
\end{frame}

\section{The Q-Learning Model}
% =============================================================================

\begin{frame}{Mathematical Formulation}
  \textbf{State space.} \(s_t = (\text{own action}_{t-1},\; \text{opponent action}_{t-1})\).\\[0.4em]
  \textbf{Action space.} \(\mathcal{A} = \{C, B\}\).\\[0.4em]
  \textbf{Reward.} \(r_t\) = payoff matrix value at round \(t\).\\[0.8em]

  \textbf{Bellman update rule:}
  \[
    Q(s_t, a_t) \;\leftarrow\; Q(s_t, a_t) + \alpha\Bigl[r_t + \gamma \max_{a'} Q(s_{t+1}, a') - Q(s_t, a_t)\Bigr]
  \]
  \vspace{0.5em}
  \textbf{\(\varepsilon\)-greedy policy.} Explore with prob.\ \(\varepsilon\), exploit with prob.\ \(1-\varepsilon\).\\
  \(\varepsilon\) decays multiplicatively: starts at 1, floors at 0.01.\\[0.8em]


\end{frame}

\section{Hyperparameter Sensitivity}
% =============================================================================

\begin{frame}{Hyperparameter Sensitivity — Summary}
  \begin{columns}[T]
    \begin{column}{0.32\textwidth}
      \textbf{\(\varepsilon\) — Exploration}\\[0.4em]
      Start at 1, decay to 0.01.\\
      Too-fast decay \(\Rightarrow\) premature exploitation of a suboptimal policy.\\
      Too-slow \(\Rightarrow\) still exploring at convergence.\\[0.4em]
      Chosen: \(\varepsilon_{\text{decay}} = 0.9999\).
    \end{column}
    \begin{column}{0.32\textwidth}
      \textbf{\(\alpha\) — Learning Rate}\\[0.4em]
      Too high: Q-values volatile, oscillating policy.\\
      Too low: converges before learning against punitive strategies.\\[0.4em]
      Chosen: \(\alpha = 0.1\), robust across all opponents.
    \end{column}
    \begin{column}{0.32\textwidth}
      \textbf{\(\gamma\) — Discount Factor}\\[0.4em]
      Most critical parameter.\\
      Theoretical threshold \(\gamma > \tfrac{1}{3}\) confirmed empirically.\\[0.4em]
      Chosen: \(\gamma = 0.9\).\\[0.4em]
      Analogy: valuing future payoffs like a long-term asset.
    \end{column}
  \end{columns}
  \vspace{1em}
  
  \textbf{Chosen hyperparameters:}
  \[
    \alpha = 0.1, \quad \gamma = 0.9, \quad \varepsilon_0 = 1.0, \quad \varepsilon_{\min} = 0.01, \quad \varepsilon_{\text{decay}} = 0.9999
  \]
\end{frame}

\begin{frame}{Key Finding — The \(\gamma\) Threshold}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      Since \(T > R\) (betrayal always beats cooperation immediately), a myopic agent always defects.\\[0.6em]
      Comparing discounted value streams against TitForTat:
      \[
        V_C = \frac{3}{1-\gamma}, \qquad V_B = 4 + \frac{\gamma}{1-\gamma}
      \]
      Cooperation is rational iff \(V_C > V_B\):
      \[
        \frac{3}{1-\gamma} > 4 + \frac{\gamma}{1-\gamma} \;\Longrightarrow\; \boxed{\gamma > \tfrac{1}{3}}
      \]
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{itemize}
        \item This threshold is \alert{exactly recovered empirically} in our simulations (Figure 5).
        \item The analogy to discounting future cash flows of a financial product is direct: cooperation is the long-term investment.
        \item Our chosen \(\gamma = 0.9\) comfortably clears the threshold.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Two Q-Learning Agents — Spontaneous Cooperation}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{itemize}
        \item Both agents start from zero knowledge.
        \item Initial phase: mutual betrayal (high \(\varepsilon\), temporary reward from defection).
        \item As \(\varepsilon\) decays, both agents \alert{sharply converge to mutual cooperation}.
        \item Cooperation rate \(\to 1\) consistently across all simulation runs.
        \item Convergence takes \(\approx 30{,}000\) turns — comparable to convergence against pre-programmed strategies, underlining the role of exploration decay.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \small
      \textbf{Why this matters.} Cooperation is not programmed — it \emph{emerges} from self-interest and repeated interaction alone.
    \end{column}
  \end{columns}
\end{frame}

\section{Conclusion}
% =============================================================================

\begin{frame}{Conclusion}
  \begin{block}{Key takeaways}
    \begin{itemize}
      \item Against non-punishing opponents, Q-learning converges to systematic defection — consistent with Nash equilibrium.
      \item Against retaliatory strategies (TFT and variants), cooperation may arise, provided \(\gamma > \tfrac{1}{3}\).
      \item Two independent Q-agents discover mutual cooperation from scratch.
      \item A minority of Q-agents outperforms majority in a hostile population.
    \end{itemize}
  \end{block}
  \vspace{0.6em}
  \textbf{Limitations.}
  Memory depth of 1 limits recognition of longer-horizon strategies.
  Non-stationarity when two Q-agents face each other breaks convergence guarantees.
  Purely self-interested reward — no other-regarding preferences.
\end{frame}
